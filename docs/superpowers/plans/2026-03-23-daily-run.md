# Daily Run 워크플로우 구현 계획

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 대시보드에서 "오늘 할 일"을 한 버튼으로 실행하고, 분석 → 발행 → 보고 3단계를 실시간으로 확인할 수 있는 Daily Run 워크플로우 구현.

**Architecture:** Supabase `daily_runs` 테이블을 큐로 사용. 대시보드(Vercel)에서 레코드 생성 → Worker(PC)가 Realtime으로 감지하여 3단계 파이프라인 실행. UI는 Realtime 구독으로 실시간 갱신.

**Tech Stack:** Next.js 16 (App Router), Supabase (Postgres + Realtime + RLS), Python Worker, OpenAI GPT-4o, shadcn/ui

```sql
-- daily_runs 테이블
CREATE TABLE daily_runs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES auth.users(id),
  status TEXT NOT NULL DEFAULT 'pending',
  mode TEXT NOT NULL DEFAULT 'auto',
  trigger_type TEXT NOT NULL DEFAULT 'manual',
  analysis JSONB,
  plan JSONB,
  report TEXT,
  todos JSONB,
  error TEXT,
  started_at TIMESTAMPTZ,
  completed_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- RLS
ALTER TABLE daily_runs ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can view own daily_runs"
  ON daily_runs FOR SELECT USING (auth.uid() = user_id);
CREATE POLICY "Users can insert own daily_runs"
  ON daily_runs FOR INSERT WITH CHECK (auth.uid() = user_id);
CREATE POLICY "Users can update own daily_runs"
  ON daily_runs FOR UPDATE USING (auth.uid() = user_id);
CREATE POLICY "Users can delete own daily_runs"
  ON daily_runs FOR DELETE USING (auth.uid() = user_id);

-- Indexes
CREATE INDEX idx_daily_runs_user_status ON daily_runs(user_id, status);
CREATE INDEX idx_daily_runs_user_created ON daily_runs(user_id, created_at DESC);

-- publish_jobs FK
ALTER TABLE publish_jobs ADD COLUMN daily_run_id UUID REFERENCES daily_runs(id);
CREATE INDEX idx_publish_jobs_daily_run ON publish_jobs(daily_run_id);

-- Realtime 활성화
ALTER PUBLICATION supabase_realtime ADD TABLE daily_runs;
```

```typescript
// src/lib/daily-run.ts
import { createClient } from '@/lib/supabase/server'

export interface DailyRun {
  id: string
  user_id: string
  status: string
  mode: string
  trigger_type: string
  analysis: Record<string, unknown> | null
  plan: Record<string, unknown> | null
  report: string | null
  todos: Array<{
    id: string
    type: string
    priority: string
    blog_id: string
    title: string
    reason: string
    done: boolean
  }> | null
  error: string | null
  started_at: string | null
  completed_at: string | null
  created_at: string
}

export interface DailyRunJob {
  id: number
  blog_id: string
  keyword: string
  status: string
  title: string | null
  published_url: string | null
  publish_error: string | null
  created_at: string
}

export async function getTodayRun(): Promise<DailyRun | null> {
  const supabase = await createClient()

  // KST 기준 오늘 날짜
  const kstNow = new Date(Date.now() + 9 * 60 * 60 * 1000)
  const today = kstNow.toISOString().split('T')[0]

  const { data } = await supabase
    .from('daily_runs')
    .select('*')
    .gte('created_at', `${today}T00:00:00+09:00`)
    .order('created_at', { ascending: false })
    .limit(1)
    .maybeSingle()

  return data
}

export async function getDailyRun(runId: string): Promise<{
  run: DailyRun
  jobs: DailyRunJob[]
} | null> {
  const supabase = await createClient()

  const { data: run } = await supabase
    .from('daily_runs')
    .select('*')
    .eq('id', runId)
    .single()

  if (!run) return null

  const { data: jobs } = await supabase
    .from('publish_jobs')
    .select('id, blog_id, keyword, status, title, published_url, publish_error, created_at')
    .eq('daily_run_id', runId)
    .order('created_at')

  return { run, jobs: jobs || [] }
}
```

```typescript
// src/app/api/daily-run/route.ts
import { createClient } from '@/lib/supabase/server'
import { NextResponse } from 'next/server'

export async function POST(request: Request) {
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

  const { mode = 'auto' } = await request.json()

  if (!['auto', 'manual'].includes(mode)) {
    return NextResponse.json({ error: 'mode는 auto 또는 manual' }, { status: 400 })
  }

  // KST 기준 오늘 중복 확인
  const kstNow = new Date(Date.now() + 9 * 60 * 60 * 1000)
  const today = kstNow.toISOString().split('T')[0]

  const { data: existing } = await supabase
    .from('daily_runs')
    .select('id, status')
    .eq('user_id', user.id)
    .gte('created_at', `${today}T00:00:00+09:00`)
    .not('status', 'in', '(completed,failed,cancelled)')
    .limit(1)

  if (existing && existing.length > 0) {
    return NextResponse.json({
      error: '이미 진행 중인 실행이 있습니다',
      runId: existing[0].id,
    }, { status: 409 })
  }

  const { data, error } = await supabase
    .from('daily_runs')
    .insert({
      user_id: user.id,
      status: 'pending',
      mode,
      trigger_type: 'manual',
    })
    .select('id')
    .single()

  if (error) return NextResponse.json({ error: error.message }, { status: 500 })

  return NextResponse.json({ runId: data.id, message: '실행 시작됨' })
}
```

```typescript
// src/components/daily-run-trigger.tsx
'use client'

import { useState } from 'react'
import { useRouter } from 'next/navigation'
import { Button } from '@/components/ui/button'

export function DailyRunTrigger() {
  const router = useRouter()
  const [state, setState] = useState<'idle' | 'selecting' | 'loading'>('idle')
  const [error, setError] = useState('')

  async function startRun(mode: 'auto' | 'manual') {
    setState('loading')
    setError('')
    try {
      const res = await fetch('/api/daily-run', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ mode }),
      })
      const data = await res.json()
      if (!res.ok) {
        if (res.status === 409 && data.runId) {
          router.push(`/daily-run/${data.runId}`)
          return
        }
        setError(data.error || '요청 실패')
        setState('idle')
        return
      }
      router.push(`/daily-run/${data.runId}`)
    } catch {
      setError('네트워크 오류')
      setState('idle')
    }
  }

  if (state === 'selecting') {
    return (
      <div className="flex gap-2">
        <Button size="sm" onClick={() => startRun('auto')} variant="default">
          자동 발행
        </Button>
        <Button size="sm" onClick={() => startRun('manual')} variant="outline">
          수동 리뷰
        </Button>
        <Button size="sm" onClick={() => setState('idle')} variant="ghost">
          취소
        </Button>
      </div>
    )
  }

  return (
    <div>
      <Button
        size="sm"
        onClick={() => setState('selecting')}
        disabled={state === 'loading'}
      >
        {state === 'loading' ? '시작 중...' : '실행하기'}
      </Button>
      {error && <p className="text-xs text-red-500 mt-1">{error}</p>}
    </div>
  )
}
```

```typescript
'use client'

import { useEffect, useState } from 'react'
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card'
import { Badge } from '@/components/ui/badge'
import { DailyRunTrigger } from '@/components/daily-run-trigger'
import { createClient } from '@/lib/supabase/client'
import type { DailyRun } from '@/lib/daily-run'

const STATUS_LABELS: Record<string, { label: string; color: string }> = {
  pending: { label: '대기 중', color: 'bg-gray-100 text-gray-700' },
  analyzing: { label: '분석 중', color: 'bg-blue-100 text-blue-700' },
  plan_ready: { label: '계획 완료', color: 'bg-indigo-100 text-indigo-700' },
  publishing: { label: '발행 중', color: 'bg-purple-100 text-purple-700' },
  reporting: { label: '보고 생성 중', color: 'bg-amber-100 text-amber-700' },
  completed: { label: '완료', color: 'bg-emerald-100 text-emerald-700' },
  failed: { label: '실패', color: 'bg-red-100 text-red-700' },
  cancelled: { label: '취소됨', color: 'bg-gray-100 text-gray-500' },
  finalize_requested: { label: '마감 중', color: 'bg-amber-100 text-amber-700' },
}

export function DailyRunCard({ initialRun }: { initialRun: DailyRun | null }) {
  const [run, setRun] = useState<DailyRun | null>(initialRun)

  useEffect(() => {
    const supabase = createClient()

    const channel = supabase
      .channel('daily-run-card')
      .on('postgres_changes', {
        event: '*',
        schema: 'public',
        table: 'daily_runs',
      }, (payload) => {
        const row = payload.new as DailyRun
        if (row) setRun(row)
      })
      .subscribe()

    return () => { supabase.removeChannel(channel) }
  }, [])

  const statusInfo = run ? STATUS_LABELS[run.status] || { label: run.status, color: '' } : null
  const todosCount = run?.todos?.length || 0
  const isTerminal = run && ['completed', 'failed', 'cancelled'].includes(run.status)

  return (
    <Card>
      <CardHeader className="pb-2">
        <div className="flex items-center justify-between">
          <CardTitle className="text-sm font-medium text-gray-500">오늘의 실행</CardTitle>
          {(!run || isTerminal) && <DailyRunTrigger />}
        </div>
      </CardHeader>
      <CardContent>
        {!run ? (
          <p className="text-sm text-gray-400">아직 오늘의 실행이 없습니다</p>
        ) : (
          <div className="space-y-2">
            <div className="flex items-center gap-2">
              <Badge className={statusInfo?.color}>{statusInfo?.label}</Badge>
              <span className="text-xs text-gray-400">
                {run.mode === 'auto' ? '자동' : '수동'} 모드
                {run.trigger_type === 'scheduled' && ' · 예약 실행'}
              </span>
            </div>
            {run.status === 'completed' && (
              <div className="text-sm text-gray-600">
                {todosCount > 0 && <span>TODO {todosCount}건 · </span>}
                <a href={`/daily-run/${run.id}`} className="text-emerald-600 hover:underline">
                  상세 보기 →
                </a>
              </div>
            )}
            {!isTerminal && run.status !== 'pending' && (
              <a href={`/daily-run/${run.id}`} className="text-sm text-emerald-600 hover:underline">
                진행 상황 보기 →
              </a>
            )}
          </div>
        )}
      </CardContent>
    </Card>
  )
}
```

```typescript
// 간단한 마크다운 → HTML 변환 (볼드, 리스트, 헤딩, 줄바꿈) + 기본 XSS 방어
function markdownToHtml(md: string): string {
  // 위험한 태그 제거
  let html = md
    .replace(/<script\b[^<]*(?:(?!<\/script>)<[^<]*)*<\/script>/gi, '')
    .replace(/<iframe\b[^>]*>.*?<\/iframe>/gi, '')
    .replace(/\bon\w+\s*=\s*"[^"]*"/gi, '')
    .replace(/\bon\w+\s*=\s*'[^']*'/gi, '')

  // 마크다운 변환
  html = html
    .replace(/### (.+)/g, '<h3>$1</h3>')
    .replace(/## (.+)/g, '<h2>$1</h2>')
    .replace(/# (.+)/g, '<h1>$1</h1>')
    .replace(/\*\*(.+?)\*\*/g, '<strong>$1</strong>')
    .replace(/^- (.+)/gm, '<li>$1</li>')
    .replace(/(<li>.*<\/li>\n?)+/g, '<ul>$&</ul>')
    .replace(/\n\n/g, '<br/><br/>')
    .replace(/\n/g, '<br/>')

  return html
}

      .channel(`daily-run-${run.id}`)
      .on('postgres_changes', {
        event: 'UPDATE',
        schema: 'public',
        table: 'daily_runs',
        filter: `id=eq.${run.id}`,
      }, (payload) => {
        setRun(payload.new as DailyRun)
      })
      .subscribe()

    const jobsChannel = supabase
      .channel(`daily-run-jobs-${run.id}`)
      .on('postgres_changes', {
        event: '*',
        schema: 'public',
        table: 'publish_jobs',
        filter: `daily_run_id=eq.${run.id}`,

          const exists = prev.find(j => j.id === updated.id)
          if (exists) return prev.map(j => j.id === updated.id ? updated : j)
          return [...prev, updated]
        })

      .subscribe()

    return () => {
      supabase.removeChannel(runChannel)
      supabase.removeChannel(jobsChannel)
    }
  }, [run.id])

    await fetch(`/api/daily-run/${run.id}/cancel`, { method: 'POST' })
```

```python
import json
import logging
import os
from datetime import datetime, timezone, timedelta

from config import get_supabase, WORKER_USER_ID

logger = logging.getLogger("daily_run")


def collect_analysis(supabase) -> dict:
    """블로그별 현황 수집"""
    blogs_result = supabase.table("blogs").select("id, label, url, platform, adapter").eq(
        "user_id", WORKER_USER_ID
    ).execute()
    blogs = blogs_result.data or []

    if not blogs:
        return {"blogs": {}, "summary": {"total_blogs": 0, "data_available": False}}

    # measurements — 최근 14일 (user_id 필터 필수 — service role은 RLS 무시)
    since_14d = (datetime.now(timezone.utc) - timedelta(days=14)).strftime("%Y-%m-%d")
    meas_result = supabase.table("measurements").select("measured_at, data").eq(
        "user_id", WORKER_USER_ID
    ).gte(
        "measured_at", since_14d
    ).order("measured_at", desc=True).execute()
    measurements = meas_result.data or []

    # keywords — 전체
    kw_result = supabase.table("keywords").select(
        "blog_id, status, priority"
    ).eq("user_id", WORKER_USER_ID).execute()
    all_keywords = kw_result.data or []

    # publish_jobs — 최근 7일
    since_7d = (datetime.now(timezone.utc) - timedelta(days=7)).isoformat()
    jobs_result = supabase.table("publish_jobs").select(
        "blog_id, status, index_status, created_at"
    ).eq("user_id", WORKER_USER_ID).gte("created_at", since_7d).execute()
    recent_jobs = jobs_result.data or []

    blog_analysis = {}
    for blog in blogs:
        bid = blog["id"]

        # 키워드 현황
        blog_kws = [k for k in all_keywords if k["blog_id"] == bid]
        pending = sum(1 for k in blog_kws if k["status"] != "published")
        published = sum(1 for k in blog_kws if k["status"] == "published")
        urgent = sum(1 for k in blog_kws if k.get("priority") == "urgent" and k["status"] != "published")
        high = sum(1 for k in blog_kws if k.get("priority") == "high" and k["status"] != "published")

        # 최근 발행
        blog_jobs = [j for j in recent_jobs if j["blog_id"] == bid]
        published_jobs = [j for j in blog_jobs if j["status"] == "published"]
        indexed = sum(1 for j in blog_jobs if j.get("index_status") == "requested")
        total_published_jobs = len(published_jobs)

        # 트래픽/수익 (measurements에서 추출)
        traffic = None
        revenue = None
        if measurements:
            # 최근 7일 vs 이전 7일 비교
            recent_7 = measurements[:7]
            prev_7 = measurements[7:14]

            def sum_field(ms, *keys, default=0):
                total = 0
                for m in ms:
                    d = m.get("data") or {}
                    val = d
                    for k in keys:
                        val = (val or {}).get(k, default)
                    total += val or 0
                return total

            # 트래픽 (measurements.data.adsense.pageviews 등에서 추출)
            recent_views = sum_field(recent_7, "adsense", "pageviews")
            prev_views = sum_field(prev_7, "adsense", "pageviews")
            if recent_views > 0 or prev_views > 0:
                view_trend = f"{((recent_views - prev_views) / max(prev_views, 1)) * 100:+.1f}%" if prev_views > 0 else "N/A"
                traffic = {
                    "recent_7d_views": int(recent_views),
                    "prev_7d_views": int(prev_views),
                    "trend": view_trend,
                }
            # 트래픽 데이터가 measurements에 없으면 traffic=None (graceful degradation)

            # 수익
            recent_rev = sum_field(recent_7, "adsense", "revenue") + sum_field(recent_7, "coupang", "revenue")
            prev_rev = sum_field(prev_7, "adsense", "revenue") + sum_field(prev_7, "coupang", "revenue")
            rev_trend = f"{((recent_rev - prev_rev) / max(prev_rev, 1)) * 100:+.1f}%" if prev_rev > 0 else "N/A"

            revenue = {
                "recent_7d_total": int(recent_rev),
                "adsense": int(sum_field(recent_7, "adsense", "revenue")),
                "coupang_clicks": int(sum_field(recent_7, "coupang", "clicks")),
                "trend": rev_trend,
            }

        blog_analysis[bid] = {
            "label": blog.get("label", bid),
            "url": blog.get("url", ""),
            "traffic": traffic,
            "revenue": revenue,
            "keywords": {
                "total": len(blog_kws),
                "pending": pending,
                "published": published,
                "urgent": urgent,
                "high": high,
            },
            "recent_posts": {
                "last_7d_count": total_published_jobs,
                "avg_daily": round(total_published_jobs / 7, 1),
            },
            "indexing": {
                "total_published": total_published_jobs,
                "indexed": indexed,
                "pending": max(0, total_published_jobs - indexed),
                "rate": f"{indexed / max(total_published_jobs, 1) * 100:.0f}%",
            },
        }

    total_pending = sum(b["keywords"]["pending"] for b in blog_analysis.values())
    total_rev = sum((b["revenue"]["recent_7d_total"] if b["revenue"] else 0) for b in blog_analysis.values())

    return {
        "blogs": blog_analysis,
        "summary": {
            "total_blogs": len(blogs),
            "total_pending_keywords": total_pending,
            "weekly_revenue": total_rev,
            "data_available": len(measurements) > 0,
        },
```

```python
import openai


def select_top_keywords(supabase, blog_id: str, count: int) -> list:
    """우선순위 기반 키워드 선택 (fallback용)"""
    result = supabase.table("keywords").select("id, keyword, priority, expected_clicks_4w").eq(
        "blog_id", blog_id
    ).eq("user_id", WORKER_USER_ID).eq("status", "pending").limit(50).execute()

    if not result.data:
        return []

    priority_order = {"urgent": 0, "high": 1, "medium": 2, "low": 3}
    sorted_kws = sorted(result.data, key=lambda k: (
        priority_order.get(k.get("priority", "medium"), 2),
        -(k.get("expected_clicks_4w") or 0),
    ))
    return sorted_kws[:count]


def fallback_plan(supabase, analysis: dict) -> dict:
    """GPT 실패 시 fallback — 블로그당 1건, 키워드 우선순위 기반"""
    plan = {"blogs": {}, "total_jobs": 0}
    for blog_id, blog_data in analysis.get("blogs", {}).items():
        if blog_data["keywords"]["pending"] > 0:
            keywords = select_top_keywords(supabase, blog_id, 1)
            if keywords:
                kw = keywords[0]
                plan["blogs"][blog_id] = {
                    "recommended_count": 1,
                    "reason": "GPT 계획 실패 — 키워드 우선순위 기반 fallback",
                    "keywords": [{
                        "keyword_id": kw["id"],
                        "keyword": kw["keyword"],
                        "priority": kw.get("priority", "medium"),
                        "expected_clicks": kw.get("expected_clicks_4w", 0),
                    }],
                }
                plan["total_jobs"] += 1
    return plan

    try:
```

```python
# worker/daily_run.py (이어서)

import asyncio


async def run_daily_workflow(run_id: str, supabase):
    """Daily Run 3단계 파이프라인"""
    logger.info(f"=== Daily Run 시작: {run_id} ===")

    # run 데이터 조회
    run_result = supabase.table("daily_runs").select("*").eq("id", run_id).single().execute()
    run = run_result.data
    if not run:
        logger.error(f"Daily Run {run_id} not found")
        return

    mode = run["mode"]

    try:
        # === 1단계: 분석 ===
        supabase.table("daily_runs").update({
            "status": "analyzing",
            "started_at": datetime.now(timezone.utc).isoformat(),
        }).eq("id", run_id).execute()

        analysis = collect_analysis(supabase)
        supabase.table("daily_runs").update({"analysis": analysis}).eq("id", run_id).execute()
        logger.info(f"  분석 완료: {analysis['summary']['total_blogs']}개 블로그")

        # === 2단계: 발행 계획 ===
        plan = generate_publish_plan(supabase, analysis)
        supabase.table("daily_runs").update({
            "status": "plan_ready",
            "plan": plan,
        }).eq("id", run_id).execute()
        logger.info(f"  계획 수립: {plan.get('total_jobs', 0)}건 발행 예정")

        # === publish_jobs 생성 ===
        job_ids = []
        for blog_id, blog_plan in plan.get("blogs", {}).items():
            for kw in blog_plan.get("keywords", []):
                # 키워드 상태 변경
                if kw.get("keyword_id"):
                    supabase.table("keywords").update(
                        {"status": "generating"}
                    ).eq("id", kw["keyword_id"]).execute()

                job_result = supabase.table("publish_jobs").insert({
                    "user_id": WORKER_USER_ID,
                    "blog_id": blog_id,
                    "keyword_id": kw.get("keyword_id"),
                    "keyword": kw["keyword"],
                    "daily_run_id": run_id,
                    "status": "generate_requested",
                    "started_at": datetime.now(timezone.utc).isoformat(),
                }).execute()

                if job_result.data:
                    job_ids.append(job_result.data[0]["id"])
                    logger.info(f"  Job 생성: {blog_id} — {kw['keyword']}")

        supabase.table("daily_runs").update({"status": "publishing"}).eq("id", run_id).execute()

        if not job_ids:
            logger.info("  발행 대상 없음, 보고 단계로")
            supabase.table("daily_runs").update({"status": "reporting"}).eq("id", run_id).execute()
            report, todos = generate_report(supabase, analysis, plan, [])
            supabase.table("daily_runs").update({
                "status": "completed",
                "report": report,
                "todos": todos,
                "completed_at": datetime.now(timezone.utc).isoformat(),
            }).eq("id", run_id).execute()
            logger.info(f"=== Daily Run 완료: {run_id} (발행 0건) ===")
            return

        # 콘텐츠 생성은 기존 Realtime/polling이 자동 처리
        # auto 모드: 생성 완료된 job을 publish_requested로 변경
        # manual 모드: 사용자가 continue API로 진행

        # 모든 job 대기 (타임아웃 포함)
        timeout = 4 * 3600 if mode == "auto" else 24 * 3600
        start_time = datetime.now(timezone.utc)

        while True:
            elapsed = (datetime.now(timezone.utc) - start_time).total_seconds()
            if elapsed > timeout:
                logger.warning(f"  타임아웃 ({timeout}s), 현재 결과로 보고 진행")
                break

            # daily_run 상태 체크 (취소/마감 확인)
            run_check = supabase.table("daily_runs").select("status").eq("id", run_id).single().execute()
            current_status = run_check.data["status"] if run_check.data else "publishing"
            if current_status in ("cancelled", "finalize_requested"):
                logger.info(f"  {current_status} 감지, 보고 단계로")
                break

            # job 상태 확인
            jobs_check = supabase.table("publish_jobs").select("id, status").in_(
                "id", job_ids
            ).execute()
            statuses = [j["status"] for j in (jobs_check.data or [])]

            # auto 모드: completed → publish_requested 자동 전환
            if mode == "auto":
                for j in (jobs_check.data or []):
                    if j["status"] == "completed":
                        supabase.table("publish_jobs").update({
                            "status": "publish_requested",
                        }).eq("id", j["id"]).eq("status", "completed").execute()

            # 모든 job이 최종 상태인지 확인
            terminal = {"published", "publish_failed", "failed"}
            if mode == "auto":
                if all(s in terminal for s in statuses):
                    break
            else:
                # manual: completed도 최종 (사용자가 발행 안 할 수 있음)
                manual_terminal = terminal | {"completed"}
                if all(s in manual_terminal for s in statuses):
                    break

            await asyncio.sleep(10)

        # === 3단계: 보고 ===
        supabase.table("daily_runs").update({"status": "reporting"}).eq("id", run_id).execute()

        report, todos = generate_report(supabase, analysis, plan, job_ids)

        supabase.table("daily_runs").update({
            "status": "completed",
            "report": report,
            "todos": todos,
            "completed_at": datetime.now(timezone.utc).isoformat(),
        }).eq("id", run_id).execute()

        logger.info(f"=== Daily Run 완료: {run_id} ===")

    except Exception as e:
        logger.error(f"Daily Run 실패: {e}")
        supabase.table("daily_runs").update({
            "status": "failed",
            "error": str(e),
            "completed_at": datetime.now(timezone.utc).isoformat(),
        }).eq("id", run_id).execute()
```
